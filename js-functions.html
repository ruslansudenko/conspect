    Конкатенация(concatination - сцепление, соединение) - это операция объединения или склеивания строк или последовательностей символов в одну строку.
    example: 
    const str1 = "Привет, ";
    const str2 = "мир!";
    const result = str1 + str2;
    console.log(result); // Вывод: "Привет, мир!"

        Math 
                // Math.floor(num) - повертає найменше ціле число,
        // менше, або яке дорівнює зазначеному числу
        console.log(Math.floor(1.7)); // 1

        // Math.ceil(num) - повертає найбільше ціле число,
        // більше, або яке дорівнює зазначеному числу.
        console.log(Math.ceil(1.2)); // 2

        // Math.round(num) - повертає значення числа,
        // округленого до найближчого цілого
        console.log(Math.round(1.2)); // 1
        console.log(Math.round(1.5)); // 2

        // Math.max(num1, num2, ...) - повертає найбільше ціле число з набору
        console.log(Math.max(20, 10, 50, 40)); // 50

        // Math.min(num1, num2, ...) - повертає найменше ціле число з набору
        console.log(Math.min(20, 10, 50, 40)); // 10

        // Math.pow(base, exponent) - піднесення до степеня
        console.log(Math.pow(2, 4)); // 16

        // Math.random() - повертає псевдовипадкове число в діапазоні [0, 1)
        console.log(Math.random()); // випадкове число між 0 і 1
        console.log(Math.random() * (10 - 1) + 1); // псевдовипадкове число від 1 до 10

        оператор continue 
        Переводит цикл на следующуи итерацию без выполнения кода в цикле 
        for (let i = 1; i<=10; i +=1) {
            if (i === 5){
                continue;
            }
            console.log(i);
        }
        выведет 1234 678910

        оператор break - прерывает цик и полностью выходит из него 
        or (let i = 1; i<=10; i +=1) {
            if (i === 5){
                break;
            }
            console.log(i);
        }
        выведет 1234

        цикл while(condition){}
         испольлзуется в тех условиях когда не извесно колличество итераций
         например при требовании от пользователя ввести пароль цикл if или for только один раз проверит 
         введенный пароль на соответствие и выдаст результат, после чего в случае если пароль не
         верный то для следующей попытки необходимо еще раз запускать if. Но в случае с while цикл
         (вопрос "введите пароль") будет запускаться до тех пор пока не будет введен правильный пароль
         let attempts = 1;
        while (PASSWORD !== userPassword){
            userPassword = prompt("wrong password. Try again");
            attempts +=1;
            if (attempts === 3){
                break;
            }
        }
            if (userPassword === PASSWORD){
                alert("Welcome!");
            }

         цикл do while 
         цикл с постусловием , т.е в отличии от цикла while в do while сначала выпллняется лдействие
         для захода на следующий круг проверяется условие 
         do {} while (condition)

        let i = 0;
        do  {
            console.log(i)
            i +=1;
        }while (i < 10);
        Этот цикл выполняется минимум 1 первый раз. Используется КРАЙНЕ редко
